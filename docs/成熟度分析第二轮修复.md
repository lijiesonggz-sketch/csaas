```typescript

const prerequisiteCluster = clusterMaturity.find((c) =>
```

```typescript
// 获取选中的选项文本
let selectedOptionText = ''
if (answer?.answer) {
  const selectedAnswer = answer.answer
  const options = question.options || []

  if (Array.isArray(selectedAnswer)) {
    // 多选题: ["opt1", "opt2"] -> "选项A; 选项B"
    const texts = selectedAnswer
      .map((optionId: string) => {
        const option = options.find((opt: any) => opt.option_id === optionId)
        return option?.text || optionId
      })
      .filter(Boolean)
    selectedOptionText = texts.join('; ')
  } else {
    // 单选题: "opt1" -> "选项A"
    const option = options.find((opt: any) => opt.option_id === selectedAnswer)
    selectedOptionText = option?.text || selectedAnswer
  }
}

questionDetails.push({
  question_id: question.question_id,
  question_text: question.question_text,
  selected_option: answer?.answer || '',
  selected_option_text: selectedOptionText,  // ✅ 包含答案文本
  score,
  level: Math.round(score),
})
```

## 技术要点

### 聚类间冲突检测逻辑

**触发条件**:
```
前置聚类成熟度 < 阈值 AND 依赖聚类成熟度 >= 阈值 → 冲突
```

**示例**:
- 前置: "组织架构与职责分配" 成熟度 = 1.6 (< 2.0)
- 依赖: "安全策略与文化" 成熟度 = 3.2 (>= 3.0)
- 结论: 冲突！组织架构不清晰却有完善的安全策略，不合理

### 关键词匹配策略

**多关键词匹配**:
```typescript
keywords: ['组织架构', '职责']
// 只要聚类名称包含任一关键词即可
// "组织架构与职责分配" ✓
// "职责管理" ✓
// "组织设计" ✓
```

**避免误匹配**:
```typescript
prerequisiteCluster.cluster_id !== dependentCluster.cluster_id
// 确保前置和依赖不是同一个聚类
```
